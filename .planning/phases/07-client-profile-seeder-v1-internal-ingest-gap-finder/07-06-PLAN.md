```
import { NextRequest, NextResponse } from 'next/server'
import { verifySession } from '../../../../lib/apiAuth'
import { getKotoIQDb } from '../../../../lib/kotoiqDb'
import { MAX_PASTED_TEXT_CHARS, CANONICAL_FIELD_NAMES } from '../../../../lib/kotoiq/profileConfig'
import { seedProfile } from '../../../../lib/kotoiq/profileSeeder'
import { computeCompleteness } from '../../../../lib/kotoiq/profileGate'
import { extractFromPastedText } from '../../../../lib/kotoiq/profileExtractClaude'
import { detectDiscrepancies } from '../../../../lib/kotoiq/profileDiscrepancy'
import { generateClarifications, recomputeClarifications } from '../../../../lib/kotoiq/profileClarifications'
import { pickClarificationChannel, forwardViaSMS, forwardViaEmail, forwardViaPortal } from '../../../../lib/kotoiq/profileChannels'
import { runFullPipeline } from '../../../../lib/builder/pipelineOrchestrator'
import type { ProvenanceRecord } from '../../../../lib/kotoiq/profileTypes'

export const runtime = 'nodejs'
export const maxDuration = 60

const URL_RE = /\/(onboard|onboarding-dashboard|clients)\/([0-9a-f]{8}-[0-9a-f]{4}-[0-9a-f]{4}-[0-9a-f]{4}-[0-9a-f]{12}|[A-Za-z0-9_-]+)/

const ALLOWED_ACTIONS = [
  'seed','get_profile','list_profile','paste_text','update_field','add_field','delete_field','reject_field',
  'add_question','launch','list_clarifications','answer_clarification','forward_to_client','add_source',
] as const

function err(status: number, error: string, extra?: Record<string, any>) {
  return NextResponse.json({ error, ...(extra || {}) }, { status })
}

function extractClientId(body: any): string | null {
  if (body.client_id && typeof body.client_id === 'string') return body.client_id
  if (body.url && typeof body.url === 'string') {
    const m = body.url.match(URL_RE)
    return m?.[2] || null
  }
  return null
}

export async function POST(req: NextRequest) {
  const session: any = await verifySession(req)
  if (!session?.verified) return err(401, 'Unauthorized')
  const agencyId: string = session.agencyId
  const userId: string = session.userId || 'operator'

  let body: any
  try { body = await req.json() } catch { return err(400, 'Invalid JSON') }
  const action = String(body?.action || '')
  if (!(ALLOWED_ACTIONS as readonly string[]).includes(action)) {
    return err(400, 'Unknown action', { allowed_actions: ALLOWED_ACTIONS })
  }

  const db = getKotoIQDb(agencyId)

  try {
    // ── seed ───────────────────────────────────────────────────────────────
    if (action === 'seed') {
      const clientId = extractClientId(body)
      if (!clientId) return err(400, 'client_id or url required')
      if (body.pasted_text && typeof body.pasted_text !== 'string') return err(400, 'pasted_text must be a string')
      if (body.pasted_text && body.pasted_text.length > MAX_PASTED_TEXT_CHARS) return err(413, `pasted_text exceeds ${MAX_PASTED_TEXT_CHARS} chars`)

      // Cross-agency existence check (T-07 link enumeration — 404 not 403)
      const { data: clientRow } = await db.raw()
        .from('clients').select('id').eq('id', clientId).eq('agency_id', agencyId).is('deleted_at', null).maybeSingle()
      if (!clientRow) return err(404, 'Client not found')

      const result = await seedProfile({
        clientId, agencyId,
        pastedText: body.pasted_text,
        pastedTextSourceLabel: body.pasted_text_source_label,
        pastedTextSourceUrl: body.pasted_text_source_url,
        forceRebuild: !!body.force_rebuild,
      })
      return NextResponse.json({ profile: result.profile, discrepancies: result.discrepancies, sources_added: result.sourcesAdded })
    }

    // ── get_profile ────────────────────────────────────────────────────────
    if (action === 'get_profile') {
      if (!body.client_id) return err(400, 'client_id required')
      const { data: profile } = await db.clientProfile.get(body.client_id)
      if (!profile) return err(404, 'Profile not found')
      const discrepancies = detectDiscrepancies((profile as any).fields || {})
      return NextResponse.json({ profile, discrepancies })
    }

    // ── list_profile ───────────────────────────────────────────────────────
    if (action === 'list_profile') {
      const { data } = await db.clientProfile.list({ launched: body.launched })
      return NextResponse.json({ profiles: data || [] })
    }

    // ── paste_text ─────────────────────────────────────────────────────────
    if (action === 'paste_text') {
      if (!body.client_id) return err(400, 'client_id required')
      if (!body.pasted_text || typeof body.pasted_text !== 'string') return err(400, 'pasted_text required')
      if (body.pasted_text.length > MAX_PASTED_TEXT_CHARS) return err(413, `pasted_text exceeds ${MAX_PASTED_TEXT_CHARS} chars`)
      const extracted = await extractFromPastedText({
        text: body.pasted_text,
        agencyId, clientId: body.client_id,
        sourceLabel: body.source_label || 'operator_paste',
        sourceUrl: body.source_url,
      })
      if (body.commit === true) {
        const result = await seedProfile({
          clientId: body.client_id, agencyId,
          pastedText: body.pasted_text,
          pastedTextSourceLabel: body.source_label,
          pastedTextSourceUrl: body.source_url,
          forceRebuild: false,
        })
        return NextResponse.json({ extracted, profile: result.profile, discrepancies: result.discrepancies })
      }
      return NextResponse.json({ extracted })
    }

    // ── update_field ───────────────────────────────────────────────────────
    if (action === 'update_field') {
      if (!body.client_id || !body.field_name) return err(400, 'client_id and field_name required')
      const { data: profile } = await db.clientProfile.get(body.client_id)
      if (!profile) return err(404, 'Profile not found')
      const record: ProvenanceRecord = {
        value: body.value ?? null,
        source_type: body.source_type === 'claude_inference' ? 'claude_inference' : 'operator_edit',
        source_snippet: body.source_snippet,
        captured_at: new Date().toISOString(),
        confidence: body.source_type === 'operator_edit' ? 1.0 : Math.max(0, Math.min(1, Number(body.confidence) || 0.85)),
        edit_history: undefined,
      }
      const { error } = await db.clientProfile.updateField((profile as any).id, body.field_name, record)
      if (error) return err(500, 'Update failed', { details: String((error as any).message || error) })
      // Fire-and-forget re-score
      ;(async () => {
        try {
          const { data: refreshed } = await db.clientProfile.get(body.client_id)
          if (refreshed) {
            const score = await computeCompleteness(refreshed as any)
            await db.clientProfile.upsert({ client_id: body.client_id, agency_id: agencyId, completeness_score: score.completeness_score, completeness_reasoning: score.completeness_reasoning, soft_gaps: score.soft_gaps } as any)
            await recomputeClarifications({ profile: { id: (profile as any).id, client_id: body.client_id, agency_id: agencyId }, softGaps: score.soft_gaps, agencyId, clientId: body.client_id })
          }
        } catch (e) { console.error('[profile.route] rescore failed', e) }
      })()
      return NextResponse.json({ ok: true })
    }

    // ── add_field ──────────────────────────────────────────────────────────
    if (action === 'add_field') {
      if (!body.client_id || !body.field_name) return err(400, 'client_id and field_name required')
      if (CANONICAL_FIELD_NAMES.includes(body.field_name)) return err(400, 'field_name already in canonical schema — use update_field', { hint: 'D-05 add_field is for NEW custom operator fields' })
      const { data: profile } = await db.clientProfile.get(body.client_id)
      if (!profile) return err(404, 'Profile not found')
      const record: ProvenanceRecord = {
        value: body.value ?? null,
        source_type: 'operator_edit',
        captured_at: new Date().toISOString(),
        confidence: 1.0,
      }
      const { error } = await db.clientProfile.addField((profile as any).id, body.field_name, body.value, record as any)
      if (error) return err(500, 'Add failed')
      return NextResponse.json({ ok: true })
    }

    // ── delete_field ───────────────────────────────────────────────────────
    if (action === 'delete_field') {
      if (!body.client_id || !body.field_name) return err(400, 'client_id and field_name required')
      const { data: profile } = await db.clientProfile.get(body.client_id)
      if (!profile) return err(404, 'Profile not found')
      const { error } = await db.clientProfile.deleteField((profile as any).id, body.field_name)
      if (error) return err(500, 'Delete failed')
      return NextResponse.json({ ok: true })
    }

    // ── reject_field (PROF-05 — rejection keeps provenance) ────────────────
    if (action === 'reject_field') {
      if (!body.client_id || !body.field_name) return err(400, 'client_id and field_name required')
      const { data: profile } = await db.clientProfile.get(body.client_id)
      if (!profile) return err(404, 'Profile not found')
      const fields = (profile as any).fields || {}
      const existing = fields[body.field_name]
      if (!existing) return err(404, 'Field not present')
      // Mark rejected — preserve records
      const patched = existing.map((r: ProvenanceRecord) => ({ ...r, rejected: true }))
      fields[body.field_name] = patched
      const isHot = CANONICAL_FIELD_NAMES.slice(0, 11).includes(body.field_name)
      const patch: any = { fields, last_edited_at: new Date().toISOString() }
      if (isHot) patch[body.field_name] = null
      const { error } = await db.clientProfile.upsert({ client_id: body.client_id, agency_id: agencyId, ...patch })
      if (error) return err(500, 'Reject failed')
      return NextResponse.json({ ok: true })
    }

    // ── add_question (operator-authored clarification — D-12) ──────────────
    if (action === 'add_question') {
      if (!body.client_id || !body.question) return err(400, 'client_id and question required')
      if (String(body.question).length > 2000) return err(400, 'question too long')
      const { data: profile } = await db.clientProfile.get(body.client_id)
      const profileId = (profile as any)?.id || null
      const sev = body.severity === 'high' || body.severity === 'medium' || body.severity === 'low' ? body.severity : 'medium'
      const { data } = await db.clarifications.create({
        client_id: body.client_id,
        profile_id: profileId,
        question: body.question,
        reason: body.reason || null,
        target_field_path: body.target_field_path || null,
        severity: sev,
        impact_hint: body.impact_hint || null,
        impact_unlocks: body.impact_unlocks || [],
      })
      return NextResponse.json({ clarification: data })
    }

    // ── launch (D-15 non-blocking launch) ──────────────────────────────────
    if (action === 'launch') {
      if (!body.client_id) return err(400, 'client_id required')
      const { data: profile } = await db.clientProfile.get(body.client_id)
      if (!profile) return err(404, 'Profile not found')

      // Refresh completeness snapshot (stored, not a gate — D-13)
      const gate = await computeCompleteness(profile as any).catch(() => ({ completeness_score: 0, completeness_reasoning: 'gate unavailable', soft_gaps: [] as any[] }))
      await db.clientProfile.upsert({ client_id: body.client_id, agency_id: agencyId, completeness_score: gate.completeness_score, completeness_reasoning: gate.completeness_reasoning, soft_gaps: gate.soft_gaps } as any).catch(() => {})

      const target_keywords = Array.isArray(body.target_keywords) ? body.target_keywords : []
      const config = {
        client_id: body.client_id,
        agency_id: agencyId,
        site_id: body.site_id,
        target_keywords,
        auto_publish: !!body.auto_publish,
        stages_to_run: body.stages_to_run,
      } as any
      const runId = await runFullPipeline(config)
      await db.clientProfile.markLaunched((profile as any).id, runId)
      return NextResponse.json({ run_id: runId, completeness_score: gate.completeness_score, soft_gaps: gate.soft_gaps })
    }

    // ── list_clarifications ────────────────────────────────────────────────
    if (action === 'list_clarifications') {
      const filters = { client_id: body.client_id, status: body.status, severity: body.severity, limit: body.limit ? Math.min(500, Number(body.limit)) : undefined }
      const { data } = await db.clarifications.list(filters as any)
      return NextResponse.json({ clarifications: data || [] })
    }

    // ── answer_clarification (PROF-05) ─────────────────────────────────────
    if (action === 'answer_clarification') {
      if (!body.clarification_id || !body.answer_text) return err(400, 'clarification_id and answer_text required')
      const { data: clar } = await db.clarifications.get(body.clarification_id)
      if (!clar) return err(404, 'Clarification not found')
      await db.clarifications.markAnswered(body.clarification_id, String(body.answer_text).slice(0, 4000), userId)
      // Propagate into profile if there's a target field
      if ((clar as any).target_field_path && body.update_field !== false) {
        const { data: profile } = await db.clientProfile.get((clar as any).client_id)
        if (profile) {
          const record: ProvenanceRecord = {
            value: String(body.answer_text),
            source_type: 'operator_edit',
            captured_at: new Date().toISOString(),
            confidence: 1.0,
          }
          await db.clientProfile.updateField((profile as any).id, (clar as any).target_field_path, record)
        }
      }
      return NextResponse.json({ ok: true })
    }

    // ── forward_to_client ──────────────────────────────────────────────────
    if (action === 'forward_to_client') {
      if (!body.clarification_id) return err(400, 'clarification_id required')
      const { data: clar } = await db.clarifications.get(body.clarification_id)
      if (!clar) return err(404, 'Clarification not found')
      const { data: clientRow } = await db.raw()
        .from('clients').select('id, name, email, phone').eq('id', (clar as any).client_id).eq('agency_id', agencyId).is('deleted_at', null).maybeSingle()
      if (!clientRow) return err(404, 'Client not found')

      // Resolve agency_name for copy — read from agencies
      const { data: agencyRow } = await db.raw().from('agencies').select('name').eq('id', agencyId).maybeSingle()
      const agencyName = (agencyRow as any)?.name || 'Koto'

      const channelArg: string = typeof body.channel === 'string' ? body.channel : 'auto'
      let channel: 'sms' | 'email' | 'portal'
      if (channelArg === 'sms' || channelArg === 'email' || channelArg === 'portal') {
        channel = channelArg
      } else {
        const picked = await pickClarificationChannel({ question: (clar as any).question, agencyId, clientId: (clar as any).client_id })
        channel = picked.channel === 'operator' ? 'portal' : picked.channel
      }

      const common = {
        clarificationId: body.clarification_id,
        clientId: (clar as any).client_id,
        agencyId,
        agencyName,
        questionText: (clar as any).question,
      }
      if (channel === 'sms') {
        if (!clientRow.phone) return err(400, 'Client has no phone — cannot SMS')
        const out = await forwardViaSMS({ ...common, clientPhone: clientRow.phone })
        if (!out.ok && /rate limit/i.test(out.error || '')) return err(429, out.error!)
        return NextResponse.json({ ok: out.ok, channel: 'sms', error: out.error })
      }
      if (channel === 'email') {
        if (!clientRow.email) return err(400, 'Client has no email — cannot email')
        const out = await forwardViaEmail({
          ...common,
          clientName: clientRow.name || '',
          clientEmail: clientRow.email,
          reason: (clar as any).reason,
          impactHint: (clar as any).impact_hint,
        })
        return NextResponse.json({ ok: out.ok, channel: 'email', error: out.error })
      }
      // portal
      const out = await forwardViaPortal(common)
      return NextResponse.json({ ok: out.ok, channel: 'portal', error: out.error })
    }

    // ── add_source (Plan 7 drop-zone stub records) ─────────────────────────
    if (action === 'add_source') {
      if (!body.client_id || !body.source_type) return err(400, 'client_id and source_type required')
      const { data: profile } = await db.clientProfile.get(body.client_id)
      if (!profile) return err(404, 'Profile not found')
      const existing = (profile as any).sources || []
      existing.push({
        source_type: body.source_type,
        source_url: body.source_url,
        source_ref: body.source_ref,
        added_at: new Date().toISOString(),
        added_by: userId,
        metadata: body.metadata || {},
      })
      await db.clientProfile.upsert({ client_id: body.client_id, agency_id: agencyId, sources: existing } as any)
      return NextResponse.json({ ok: true })
    }

    return err(400, 'Unhandled action')
  } catch (e: any) {
    console.error('[profile.route] error', e)
    return err(500, 'Internal error', { message: String(e?.message || e).slice(0, 500) })
  }
}
```
