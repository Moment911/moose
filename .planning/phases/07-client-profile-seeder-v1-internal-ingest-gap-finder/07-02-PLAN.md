import 'server-only'
import { createClient } from '@supabase/supabase-js'
import { pick } from '../kotoClientPick'
import type { ProvenanceRecord, SourceType } from './profileTypes'

const SUPABASE_URL = process.env.NEXT_PUBLIC_SUPABASE_URL!
const SERVICE_KEY  = process.env.SUPABASE_SERVICE_ROLE_KEY!
const APP_URL      = process.env.NEXT_PUBLIC_APP_URL || 'https://hellokoto.com'

function sb() {
  return createClient(SUPABASE_URL, SERVICE_KEY, {
    auth: { persistSession: false },
  })
}

// Shape returned by Supabase for the clients table (partial — only fields we read).
export type ClientsRow = Record<string, any>
export type DiscoveryRow = Record<string, any>
export type RecipientRow = Record<string, any>

type Ctx = { clientId: string; agencyId: string; nowIso?: string }

function nowIsoOr(s?: string) {
  return s || new Date().toISOString()
}

function makeRec(partial: Partial<ProvenanceRecord> & { value: ProvenanceRecord['value'] }): ProvenanceRecord {
  return {
    source_type: partial.source_type ?? 'onboarding_form',
    captured_at: partial.captured_at ?? new Date().toISOString(),
    confidence: partial.confidence ?? 0.8,
    ...partial,
    value: partial.value,
  } as ProvenanceRecord
}

// ─────────────────────────────────────────────────────────────────────────
// 1. pullFromClient — clients table (dedicated columns + onboarding_answers jsonb)
//    Mapping verbatim from RESEARCH §3.1
// ─────────────────────────────────────────────────────────────────────────
export async function pullFromClient(ctx: Ctx): Promise<{ client: ClientsRow | null; records: Record<string, ProvenanceRecord[]> }> {
  const { data: client, error } = await sb()
    .from('clients')
    .select('*')
    .eq('id', ctx.clientId)
    .eq('agency_id', ctx.agencyId)   // MANDATORY cross-agency guard (RESEARCH §15 T-07-01)
    .is('deleted_at', null)
    .maybeSingle()
  if (error || !client) return { client: null, records: {} }

  const sourceUrl = `${APP_URL}/clients/${ctx.clientId}`
  const captured = nowIsoOr(client.updated_at)

  const addIf = (out: Record<string, ProvenanceRecord[]>, field: string, value: any, confidence: number, source_type: SourceType = 'onboarding_form') => {
    if (value === null || value === undefined || value === '') return
    const rec = makeRec({ value, source_type, source_url: sourceUrl, captured_at: captured, confidence })
    out[field] = out[field] || []
    out[field].push(rec)
  }

  const records: Record<string, ProvenanceRecord[]> = {}
  addIf(records, 'business_name',       client.name,                               1.0)
  addIf(records, 'website',             pick(client, 'website'),                   0.95)
  addIf(records, 'phone',               pick(client, 'phone','primary_phone'),     0.9)
  addIf(records, 'founding_year',       toIntOrNull(pick(client, 'founding_year','year_founded','business_age')), 0.8)
  addIf(records, 'primary_service',     pick(client, 'primary_service','products_services','top_services'), 0.85)
  addIf(records, 'target_customer',     pick(client, 'target_customer','ideal_customer_desc','customer_types'), 0.85)
  addIf(records, 'unique_selling_prop', pick(client, 'unique_selling_prop','why_choose_you'), 0.8)
  addIf(records, 'industry',            pick(client, 'industry'),                  0.9)
  addIf(records, 'city',                pick(client, 'city','primary_city'),       0.95)
  addIf(records, 'state',               pick(client, 'state'),                     0.95)
  addIf(records, 'service_area',        pick(client, 'service_area','service_areas','geographic_focus'), 0.8)
  addIf(records, 'marketing_budget',    pick(client, 'marketing_budget','monthly_ad_budget','budget_for_agency'), 0.8)
  addIf(records, 'welcome_statement',   pick(client, 'welcome_statement','business_description'), 0.7)
  addIf(records, 'customer_pain_points',pick(client, 'customer_pain_points'),      0.75)
  addIf(records, 'current_channels',    pick(client, 'marketing_channels','current_ad_platforms'), 0.7)

  // Voice-derived rollup that lives on clients (pre-existing fields — see _knowledge/modules/voice-onboarding.md)
  if (client.onboarding_call_summary) {
    addIf(records, 'welcome_statement', client.onboarding_call_summary, 0.8, 'voice_call')
  }

  // Per-field voice confidence scores override defaults where present
  const voiceConf = (client.onboarding_confidence_scores || {}) as Record<string, number>
  for (const [field, arr] of Object.entries(records)) {
    if (voiceConf[field] && arr[0]) arr[0].confidence = Math.max(arr[0].confidence, voiceConf[field])
  }

  return { client, records }
}

function toIntOrNull(s: string | null): number | null {
  if (s === null) return null
  const n = parseInt(String(s).replace(/[^0-9-]/g, ''), 10)
  return Number.isFinite(n) ? n : null
}

// ─────────────────────────────────────────────────────────────────────────
// 2. pullFromRecipients — koto_onboarding_recipients.answers jsonb per recipient
//    (multi-recipient support — RESEARCH §17 Risk #3)
// ─────────────────────────────────────────────────────────────────────────
export async function pullFromRecipients(ctx: Ctx): Promise<Record<string, ProvenanceRecord[]>> {
  const { data: recipients } = await sb()
    .from('koto_onboarding_recipients')
    .select('*')
    .eq('client_id', ctx.clientId)
    .eq('agency_id', ctx.agencyId)
    .order('last_active_at', { ascending: false })
  if (!recipients || recipients.length === 0) return {}

  const out: Record<string, ProvenanceRecord[]> = {}
  const sourceUrl = `${APP_URL}/onboarding-dashboard/${ctx.clientId}`

  for (const r of recipients as RecipientRow[]) {
    const answers = (r.answers || {}) as Record<string, any>
    const captured = nowIsoOr(r.last_active_at || r.updated_at || r.created_at)
    // Skip the _call_analysis key — that is pullFromVoiceCallAnalysis territory
    for (const [key, value] of Object.entries(answers)) {
      if (key === '_call_analysis' || key.startsWith('_')) continue
      if (value === null || value === undefined || value === '') continue
      const strValue = Array.isArray(value) ? value.join(', ') : String(value).trim()
      if (!strValue) continue
      // Normalise common onboarding-answer keys to canonical names where we know the mapping
      const canonical = mapOnboardingKeyToCanonical(key)
      if (!canonical) continue
      const rec = makeRec({
        value: strValue,
        source_type: 'onboarding_form',
        source_url: sourceUrl,
        source_ref: `recipient:${r.id}`,
        source_snippet: strValue.slice(0, 240),
        captured_at: captured,
        confidence: 0.8,
      })
      out[canonical] = out[canonical] || []
      out[canonical].push(rec)
    }
  }
  return out
}

function mapOnboardingKeyToCanonical(key: string): string | null {
  const map: Record<string, string> = {
    products_services: 'primary_service',
    top_services: 'primary_service',
    ideal_customer_desc: 'target_customer',
    customer_types: 'target_customer',
    why_choose_you: 'unique_selling_prop',
    monthly_ad_budget: 'marketing_budget',
    budget_for_agency: 'marketing_budget',
    business_description: 'welcome_statement',
    marketing_channels: 'current_channels',
    current_ad_platforms: 'current_channels',
    geographic_focus: 'service_area',
    service_areas: 'service_area',
    founding_year: 'founding_year',
    year_founded: 'founding_year',
    business_age: 'founding_year',
    competitors: 'competitors',
    differentiators: 'differentiators',
    pain_points: 'pain_points',
    trust_anchors: 'trust_anchors',
  }
  return map[key] || null
}

// ─────────────────────────────────────────────────────────────────────────
// 3. pullFromDiscovery — koto_discovery_engagements narrative-level fields
//    Section-level extraction (Haiku per-section) lives in Plan 3.
// ─────────────────────────────────────────────────────────────────────────
export async function pullFromDiscovery(ctx: Ctx): Promise<{ engagement: DiscoveryRow | null; narrativeRecords: Record<string, ProvenanceRecord[]> }> {
  const { data: engagement } = await sb()
    .from('koto_discovery_engagements')
    .select('*')
    .eq('client_id', ctx.clientId)
    .eq('agency_id', ctx.agencyId)
    .order('created_at', { ascending: false })
    .limit(1)
    .maybeSingle()
  if (!engagement) return { engagement: null, narrativeRecords: {} }

  const sourceUrl = `${APP_URL}/discovery/${engagement.id}`
  const captured = nowIsoOr(engagement.updated_at || engagement.created_at)
  const out: Record<string, ProvenanceRecord[]> = {}

  if (engagement.executive_summary) {
    out.welcome_statement = [makeRec({
      value: String(engagement.executive_summary).trim(),
      source_type: 'discovery_doc',
      source_url: sourceUrl,
      source_ref: `discovery_doc:${engagement.id}:executive_summary`,
      source_snippet: String(engagement.executive_summary).slice(0, 240),
      captured_at: captured,
      confidence: 0.85,
    })]
  }

  const clientAnswers = (engagement.client_answers || {}) as Record<string, any>
  for (const [key, value] of Object.entries(clientAnswers)) {
    const canonical = mapOnboardingKeyToCanonical(key)
    if (!canonical) continue
    const strValue = Array.isArray(value) ? value.join(', ') : String(value ?? '').trim()
    if (!strValue) continue
    out[canonical] = out[canonical] || []
    out[canonical].push(makeRec({
      value: strValue,
      source_type: 'discovery_doc',
      source_url: sourceUrl,
      source_ref: `discovery_doc:${engagement.id}:client_answers.${key}`,
      source_snippet: strValue.slice(0, 240),
      captured_at: captured,
      confidence: 0.8,
    }))
  }

  return { engagement, narrativeRecords: out }
}

// ─────────────────────────────────────────────────────────────────────────
// 4. pullFromVoiceCallAnalysis — koto_onboarding_recipients.answers._call_analysis
//    (pre-existing structured Haiku output — RESEARCH §3.4 mapping)
// ─────────────────────────────────────────────────────────────────────────
export async function pullFromVoiceCallAnalysis(ctx: Ctx): Promise<Record<string, ProvenanceRecord[]>> {
  const { data: recipients } = await sb()
    .from('koto_onboarding_recipients')
    .select('id, answers, updated_at, created_at')
    .eq('client_id', ctx.clientId)
    .eq('agency_id', ctx.agencyId)
  if (!recipients || recipients.length === 0) return {}

  const out: Record<string, ProvenanceRecord[]> = {}
  for (const r of recipients as RecipientRow[]) {
    const analysis = (r.answers && r.answers._call_analysis) as Record<string, any> | undefined
    if (!analysis) continue
    const captured = analysis.analyzed_at || r.updated_at || r.created_at || new Date().toISOString()
    const callId = analysis.call_id || r.id
    const ref = `retell_call:${callId}`

    const add = (field: string, value: any, confidence: number, snippet?: string) => {
      if (value === null || value === undefined || value === '' || (Array.isArray(value) && value.length === 0)) return
      out[field] = out[field] || []
      out[field].push(makeRec({
        value: Array.isArray(value) ? value.map(String) : String(value),
        source_type: 'voice_call',
        source_ref: ref,
        source_snippet: snippet || (Array.isArray(value) ? value.join(' · ').slice(0, 240) : String(value).slice(0, 240)),
        captured_at: captured,
        confidence,
      }))
    }

    add('caller_sentiment',    analysis.caller_sentiment,       1.0)
    add('follow_up_flag',      analysis.follow_up_recommended,  1.0)
    add('expansion_signals',   analysis.upsell_signals,         0.7)
    add('pain_point_emphasis', analysis.hot_lead_reasons,       0.75)
    add('pain_point_emphasis', analysis.notable_insights,       0.7)
    if (analysis.call_summary) {
      add('welcome_statement', analysis.call_summary, 0.85, analysis.call_summary)
    }
  }
  return out
}
