```jsx
import { R, AMB } from '../../../lib/theme'

/**
 * UI-SPEC §5.11 hotspot dot.
 * Props:
 *   severity: 'low' | 'medium' | 'high'
 *   count?: number      // badge count when >1
 *   onClick(): void
 *   size?: number       // default 6px
 *   pulse?: boolean     // forces on/off; default true for high severity
 */
export default function HotspotDot({ severity = 'low', count, onClick, size = 6, pulse }) {
  const color = severity === 'high' ? R : severity === 'medium' ? AMB : '#9ca3af'
  const prefersReducedMotion = typeof window !== 'undefined' && window.matchMedia('(prefers-reduced-motion: reduce)').matches
  const shouldPulse = (pulse ?? (severity === 'high')) && !prefersReducedMotion

  return (
    <button
      type="button"
      onClick={onClick}
      aria-label={`${count || 1} open question${count && count > 1 ? 's' : ''} about this field.`}
      style={{
        all: 'unset', cursor: 'pointer', position: 'absolute',
        width: size, height: size, borderRadius: size / 2, background: color,
        animation: shouldPulse ? 'kotoHotspotPulse 1.5s ease-in-out infinite' : undefined,
      }}
    >
      {count && count > 1 && (
        <span
          style={{
            position: 'absolute', top: -6, right: -10,
            fontSize: 9, fontWeight: 700, color, background: '#fff',
            borderRadius: 6, padding: '0 3px', border: `1px solid ${color}`,
            lineHeight: 1.2,
          }}
        >{count}</span>
      )}
    </button>
  )
}
```

```jsx
const SEVERITY_PILL = {

  medium: { bg: `${AMB}1a`, color: AMB,       label: 'Would help' },
  high:   { bg: `${R}1a`,   color: R,         label: 'Blocker' },
}

function relativeTime(iso) {
  if (!iso) return ''
  const then = new Date(iso).getTime()
  const diff = Date.now() - then
  const mins = Math.floor(diff / 60000)
  if (mins < 1) return 'just now'
  if (mins < 60) return `${mins}m ago`
  const hrs = Math.floor(mins / 60)
  if (hrs < 24) return `${hrs}h ago`
  return `${Math.floor(hrs / 24)}d ago`
}
```

```jsx
import { useEffect, useState, useRef } from 'react'
import { createPortal } from 'react-dom'
import { supabase } from '../../../lib/supabase'
import HotspotDot from './HotspotDot'
import ClarificationCard from './ClarificationCard'

/**
 * UI-SPEC §5.11 + §5.8c.
 * Mounts HotspotDots on editable spans matching [data-field-path="{field}"].
 * Clicking a hotspot opens an inline ClarificationCard just below the paragraph containing the span.
 *
 * Props:
 *   clientId, agencyId, onAnswer(id, text), onForward(id, channel?), onSkip(id)
 */
export default function ClarificationsOverlay({ clientId, agencyId, onAnswer, onForward, onSkip }) {
  const [clarifications, setClarifications] = useState([])
  const [modalTarget, setModalTarget] = useState(null)   // HIGH-severity 12s sheet
  const [inlineTarget, setInlineTarget] = useState(null) // open inline card
  const [spans, setSpans] = useState([])                 // { fieldPath, rect } per matched span
  const highSeen = useRef(new Set())

  // Subscribe to clarifications
  useEffect(() => {
    if (!clientId) return
    let mounted = true
    const load = async () => {
      const res = await fetch('/api/kotoiq/profile', {
        method: 'POST', headers: { 'content-type': 'application/json' },
        body: JSON.stringify({ action: 'list_clarifications', client_id: clientId, status: 'open' }),
      })
      const j = await res.json()
      if (mounted) setClarifications(j.clarifications || [])
    }
    load()
    const ch = supabase
      .channel(`kotoiq_clarifications:${clientId}`)
      .on('postgres_changes',
        { event: '*', schema: 'public', table: 'kotoiq_clarifications', filter: `client_id=eq.${clientId}` },
        (payload) => {
          if (!mounted) return
          setClarifications(prev => {
            const arr = [...prev]
            if (payload.eventType === 'INSERT') {
              arr.push(payload.new)
              // HIGH severity + first appearance → modal escalation (D-20)
              if (payload.new.severity === 'high' && !highSeen.current.has(payload.new.id)) {
                highSeen.current.add(payload.new.id)
                setModalTarget(payload.new)
                setTimeout(() => setModalTarget(m => m?.id === payload.new.id ? null : m), 12000)
              }
            } else if (payload.eventType === 'UPDATE') {
              const i = arr.findIndex(c => c.id === payload.new.id)
              if (i >= 0) arr[i] = payload.new
              // Hide rows that left status='open'
              return arr.filter(c => c.status === 'open')
            } else if (payload.eventType === 'DELETE') {
              return arr.filter(c => c.id !== payload.old.id)
            }
            return arr.filter(c => c.status === 'open')
          })
        })
      .subscribe()
    return () => { mounted = false; supabase.removeChannel(ch) }
  }, [clientId])

  // Re-scan DOM for spans each time clarifications change (or window resizes)
  useEffect(() => {
    const update = () => {
      const nodes = Array.from(document.querySelectorAll('[data-field-path]'))
      setSpans(nodes.map(el => ({
        fieldPath: el.getAttribute('data-field-path'),
        rect: el.getBoundingClientRect(),
      })))
    }
    update()
    window.addEventListener('resize', update)
    // Capture-phase scroll listener fires for ANY scrolling ancestor (including KotoIQShellPage's
    // scroll container) — this is what makes HotspotDot stay anchored when scrolling inside a
    // scrollable shell instead of the window (fix per UI-SPEC §5.11 + v1 QA).
    window.addEventListener('scroll', update, true)
    // Re-scan after 300ms to catch layout settling
    const t = setTimeout(update, 300)
    return () => { window.removeEventListener('resize', update); window.removeEventListener('scroll', update, true); clearTimeout(t) }
  }, [clarifications])

  // Group clarifications by target_field_path and pick worst severity per field
  const byField = new Map()
  for (const c of clarifications) {
    if (!c.target_field_path) continue
    const existing = byField.get(c.target_field_path)
    if (!existing) byField.set(c.target_field_path, [c])
    else existing.push(c)
  }

  const hotspots = spans.flatMap(s => {
    const group = byField.get(s.fieldPath)
    if (!group || group.length === 0) return []
    const worst = group.reduce((acc, c) => {
      const rank = (x) => x === 'high' ? 3 : x === 'medium' ? 2 : 1
      return rank(c.severity) > rank(acc.severity) ? c : acc
    }, group[0])
    return [{ ...s, clarifications: group, worst }]
  })

  return (
    <>
      {/* HotspotDots placed via portal so they render above everything */}
      {typeof document !== 'undefined' && createPortal(
        hotspots.map(h => (
          <div
            key={h.fieldPath}
            style={{
              // position: 'fixed' is relative to the viewport, so we use the raw getBoundingClientRect
              // values directly — adding window.scrollY/scrollX would double-count scroll offset
              // when the briefing is inside a scrollable shell (fix per UI-SPEC §5.11 hotspot anchoring).
```

```jsx
export default function ClarificationsTab({ agencyId, clientId }) {
  const [list, setList] = useState([])
  const [loading, setLoading] = useState(true)
  const [severityFilter, setSeverityFilter] = useState(new Set(['low', 'medium', 'high']))
  const [statusFilter, setStatusFilter] = useState('open')
  const [ageFilter, setAgeFilter] = useState('all')    // '<1h' | 'today' | 'this_week' | 'all'
  const [selected, setSelected] = useState(new Set())

  const reload = async () => {
    setLoading(true)
    const res = await fetch('/api/kotoiq/profile', {
      method: 'POST', headers: { 'content-type': 'application/json' },
      body: JSON.stringify({ action: 'list_clarifications', client_id: clientId, status: statusFilter !== 'all' ? statusFilter : undefined }),
    })
    const j = await res.json()
    setList(j.clarifications || [])
    setLoading(false)
  }

  useEffect(() => { reload() }, [clientId, statusFilter])

  const filtered = useMemo(() => {
    const now = Date.now()
    const ageMs = { '<1h': 3600_000, 'today': 86400_000, 'this_week': 7 * 86400_000, 'all': Infinity }[ageFilter]
    return list
      .filter(c => severityFilter.has(c.severity))
      .filter(c => (now - new Date(c.created_at).getTime()) <= ageMs)
      .sort((a, b) => {
        const rank = (x) => x.severity === 'high' ? 3 : x.severity === 'medium' ? 2 : 1
        const sd = rank(b) - rank(a)
        if (sd !== 0) return sd
        return new Date(a.created_at).getTime() - new Date(b.created_at).getTime()
      })
  }, [list, severityFilter, ageFilter])

  const toggleSeverity = (s) => {
    const next = new Set(severityFilter)
    if (next.has(s)) next.delete(s); else next.add(s)
    setSeverityFilter(next)
  }

  const onAnswer = async (id, text) => {
    await fetch('/api/kotoiq/profile', { method: 'POST', headers: { 'content-type': 'application/json' }, body: JSON.stringify({ action: 'answer_clarification', clarification_id: id, answer_text: text }) })
    reload()
  }
  const onForward = async (id, channel) => {
    await fetch('/api/kotoiq/profile', { method: 'POST', headers: { 'content-type': 'application/json' }, body: JSON.stringify({ action: 'forward_to_client', clarification_id: id, channel }) })
    reload()
  }
  const onSkip = async (id) => {
    await fetch('/api/kotoiq/profile', { method: 'POST', headers: { 'content-type': 'application/json' }, body: JSON.stringify({ action: 'answer_clarification', clarification_id: id, answer_text: '[skipped]', update_field: false }) })
    reload()
  }

  const bulkForward = async () => {
    for (const id of selected) await onForward(id)
    setSelected(new Set())
  }

  if (loading) {
    return <div style={{ padding: 40, fontFamily: FB, color: '#6b7280' }}>Loading clarifications...</div>
  }
  if (filtered.length === 0) {
    return (
      <div style={{ padding: 80, textAlign: 'center', fontFamily: FB, color: BLK }}>
        <CheckCircle2 size={48} color={GRN} style={{ margin: '0 auto 16px', opacity: 0.7 }} />
        <div style={{ fontFamily: FH, fontSize: 20, fontWeight: 700, marginBottom: 8 }}>Nothing to clarify right now.</div>
        <div style={{ fontSize: 14, color: '#6b7280' }}>The pipeline is running. I'll queue questions here as they come up.</div>
      </div>
    )
  }

  return (

      {/* Filters */}
      <div style={{ display: 'flex', gap: 8, marginBottom: 24, flexWrap: 'wrap', alignItems: 'center' }}>
        {['high','medium','low'].map(s => (
          <button
            key={s}
            onClick={() => toggleSeverity(s)}
            style={{
              all: 'unset', cursor: 'pointer', padding: '4px 12px', borderRadius: 12,
              fontFamily: FH, fontSize: 12, fontWeight: 700,
              background: severityFilter.has(s) ? '#111' : '#f3f4f6',
              color: severityFilter.has(s) ? '#fff' : '#6b7280',
            }}
          >
            {s === 'high' ? 'Blocker' : s === 'medium' ? 'Would help' : 'Nice to have'}
          </button>
        ))}
        <span style={{ borderLeft: '1px solid #e5e7eb', height: 24 }} />
        {['<1h','today','this_week','all'].map(a => (
          <button
            key={a}
            onClick={() => setAgeFilter(a)}
            style={{
              all: 'unset', cursor: 'pointer', padding: '4px 10px', borderRadius: 12,
              fontFamily: FH, fontSize: 11, fontWeight: 700,
              background: ageFilter === a ? '#111' : '#f9fafb',
              color: ageFilter === a ? '#fff' : '#6b7280',
            }}
          >{a.replace('_', ' ').replace('<1h', '< 1h')}</button>
        ))}
        <span style={{ borderLeft: '1px solid #e5e7eb', height: 24 }} />
        <select value={statusFilter} onChange={e => setStatusFilter(e.target.value)} style={{ fontFamily: FB, fontSize: 12, padding: '4px 8px', borderRadius: 6, border: '1px solid #e5e7eb' }}>
          <option value="open">open</option>
          <option value="asked_client">asked client</option>
          <option value="answered">answered</option>
          <option value="skipped">skipped</option>
          <option value="all">all</option>
        </select>
        {selected.size > 0 && (

        )}
      </div>

      {filtered.map(c => (
        <div key={c.id} style={{ display: 'flex', alignItems: 'flex-start', gap: 8, marginBottom: 14 }}>
          <input
            type="checkbox"
            aria-label={`select clarification ${c.id}`}

              if (e.target.checked) next.add(c.id); else next.delete(c.id)
              setSelected(next)

            style={{ marginTop: 22 }}
          />
          <div style={{ flex: 1 }}>
            <ClarificationCard clarification={c} variant="dashboard" onAnswer={onAnswer} onForward={onForward} onSkip={onSkip} />
          </div>
        </div>
      ))}
    </div>
  )
}
```

```jsx
import { T, BLK, FH, FB } from '../../../lib/theme'

/**
 * UI-SPEC §4.1 tertiary CTA. Renders as a ghost button that opens a compact composer.
 * Props: clientId, onAdd({ question, target_field_path?, severity })
 */
export default function AskOwnQuestion({ onAdd }) {
  const [open, setOpen] = useState(false)
  const [question, setQuestion] = useState('')
  const [severity, setSeverity] = useState('medium')
  const [targetField, setTargetField] = useState('')
  const [pending, setPending] = useState(false)

  const submit = async () => {
    if (!question.trim()) return
    setPending(true)
    try {
      await onAdd?.({ question: question.trim(), target_field_path: targetField.trim() || null, severity })
      setQuestion(''); setTargetField(''); setSeverity('medium'); setOpen(false)
    } finally { setPending(false) }
  }

  if (!open) {
    return (
      <button
        onClick={() => setOpen(true)}
        style={{
          all: 'unset', cursor: 'pointer',
          fontFamily: FH, fontSize: 13, fontWeight: 700, color: T,
          padding: '8px 0',
        }}
      >
        + Ask your own question
      </button>
    )
  }

  return (
    <div style={{ padding: 16, borderRadius: 10, border: '1px solid #e5e7eb', maxWidth: 480, margin: '12px 0', fontFamily: FB, color: BLK }}>
      <textarea
        value={question}
        onChange={e => setQuestion(e.target.value)}
        placeholder="Your question — e.g. 'What's the winter discount structure?'"
        rows={2}
        maxLength={2000}
        aria-label="Clarification question"
        style={{ width: '100%', padding: 10, borderRadius: 8, border: '1px solid #e5e7eb', fontSize: 13, fontFamily: FB, resize: 'vertical', marginBottom: 8 }}
      />
      <div style={{ display: 'flex', gap: 8, alignItems: 'center', marginBottom: 8 }}>
        <input
          value={targetField}
          onChange={e => setTargetField(e.target.value)}
          placeholder="Target field (optional)"
          aria-label="Target field name"
          style={{ flex: 1, height: 32, padding: '0 10px', borderRadius: 6, border: '1px solid #e5e7eb', fontSize: 12, fontFamily: FB }}
        />
        <select value={severity} onChange={e => setSeverity(e.target.value)} style={{ height: 32, padding: '0 10px', borderRadius: 6, border: '1px solid #e5e7eb', fontSize: 12, fontFamily: FB }}>
          <option value="low">Nice to have</option>
          <option value="medium">Would help</option>
          <option value="high">Blocker</option>
        </select>
      </div>
      <div style={{ display: 'flex', gap: 8, justifyContent: 'flex-end' }}>
        <button onClick={() => { setOpen(false); setQuestion('') }} disabled={pending} style={{ all: 'unset', cursor: 'pointer', color: '#6b7280', fontSize: 12, padding: '6px 10px' }}>Cancel</button>
        <button onClick={submit} disabled={pending || !question.trim()} style={{ height: 30, padding: '0 14px', borderRadius: 6, background: T, color: '#fff', border: 'none', fontFamily: FH, fontSize: 12, fontWeight: 700, cursor: pending ? 'wait' : 'pointer', opacity: (pending || !question.trim()) ? 0.6 : 1 }}>Add question</button>
      </div>
    </div>
  )
}
```
