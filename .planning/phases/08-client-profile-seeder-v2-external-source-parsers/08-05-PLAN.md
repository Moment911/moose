**Step 1 — `src/lib/kotoiq/profileWebsiteSSRFGuard.ts`**:

```ts
import 'server-only'
import { promises as dns } from 'node:dns'
import ipaddr from 'ipaddr.js'

export class SSRFError extends Error {
  constructor(public code: 'SSRF_BAD_SCHEME' | 'SSRF_BAD_PORT' | 'SSRF_INTERNAL_IP' | 'SSRF_DNS_FAIL', msg: string) {
    super(msg)
    this.name = 'SSRFError'
  }
}

const ALLOWED_PROTOCOLS = new Set(['http:', 'https:'])
const ALLOWED_PORTS = new Set(['', '80', '443'])

const PRIVATE_RANGES = new Set([
  'private', 'loopback', 'linkLocal', 'uniqueLocal',
  'carrierGradeNat', 'multicast', 'unspecified', 'reserved', 'broadcast',
])

export type SSRFArgs = { allowExtraPorts?: boolean }

/** Refuse any URL whose resolved IP is in a private range or whose scheme/port is suspicious.
 *  Call BEFORE any fetch. Throws SSRFError on any refusal. */
export async function refuseIfInternalIp(url: string, args: SSRFArgs = {}): Promise<void> {
  let u: URL
  try { u = new URL(url) }
  catch { throw new SSRFError('SSRF_BAD_SCHEME', `Malformed URL: ${url}`) }

  if (!ALLOWED_PROTOCOLS.has(u.protocol)) {
    throw new SSRFError('SSRF_BAD_SCHEME', `Refused scheme: ${u.protocol}`)
  }
  if (!args.allowExtraPorts && !ALLOWED_PORTS.has(u.port)) {
    throw new SSRFError('SSRF_BAD_PORT', `Refused port: ${u.port}`)
  }

  // Resolve — all IPs must be public.
  let addresses: Array<{ address: string; family: number }>
  try {
    addresses = await dns.lookup(u.hostname, { all: true, verbatim: true })
  } catch (err: any) {
    throw new SSRFError('SSRF_DNS_FAIL', `DNS failed for ${u.hostname}: ${err?.code ?? err?.message}`)
  }
  if (!addresses.length) throw new SSRFError('SSRF_DNS_FAIL', `No addresses for ${u.hostname}`)

  for (const { address } of addresses) {
    let parsed
    try { parsed = ipaddr.parse(address) }
    catch { throw new SSRFError('SSRF_INTERNAL_IP', `Unparseable IP: ${address}`) }
    const range = parsed.range()
    if (PRIVATE_RANGES.has(range)) {
      throw new SSRFError('SSRF_INTERNAL_IP', `Refused range '${range}' for ${u.hostname} (${address})`)
    }
  }
}
```

**Step 2 — `src/lib/kotoiq/profileWebsiteRobots.ts`**:

```ts
import 'server-only'
import { refuseIfInternalIp } from './profileWebsiteSSRFGuard'

export type RobotsRules = {
  allow_all: boolean
  disallowed: string[]          // list of path prefixes the default user-agent is disallowed from
  explicit_allows: string[]     // Allow: rules overriding Disallow
}

export type RobotsMode = 'strict' | 'ignore' | 'warn_but_allow'

/** Fetch /robots.txt for the given site. Returns {allow_all:true, disallowed:[], explicit_allows:[]}
 *  on 404 / fetch failure (absent robots = unrestricted per RFC 9309). */
export async function fetchRobots(originUrl: string): Promise<RobotsRules> {
  const u = new URL(originUrl)
  const robotsUrl = `${u.origin}/robots.txt`
  try {
    await refuseIfInternalIp(robotsUrl)
    const r = await fetch(robotsUrl, { redirect: 'manual' })
    if (!r.ok) return { allow_all: true, disallowed: [], explicit_allows: [] }
    const body = await r.text()
    return parseRobots(body)
  } catch {
    return { allow_all: true, disallowed: [], explicit_allows: [] }
  }
}

export function parseRobots(body: string): RobotsRules {
  const lines = body.split('\n').map(l => l.replace(/#.*/, '').trim())
  let inDefaultGroup = false
  const disallowed: string[] = []
  const explicit_allows: string[] = []
  let anyRules = false
  for (const line of lines) {
    if (!line) { inDefaultGroup = false; continue }
    const m = line.match(/^([A-Za-z-]+)\s*:\s*(.*)$/)
    if (!m) continue
    const key = m[1].toLowerCase()
    const val = m[2].trim()
    if (key === 'user-agent') {
      inDefaultGroup = val === '*' || val.toLowerCase() === 'kotobot'
      continue
    }
    if (!inDefaultGroup) continue
    if (key === 'disallow') { if (val) { disallowed.push(val); anyRules = true } else { anyRules = true } }
    if (key === 'allow') { if (val) { explicit_allows.push(val); anyRules = true } }
  }
  const allow_all = !anyRules || disallowed.length === 0
  return { allow_all, disallowed, explicit_allows }
}

export type RobotsDecision = { allowed: boolean; warnings: string[] }

/** Apply mode-aware decision: strict refuses, ignore passes, warn_but_allow passes with warnings. */
export function isAllowedForCrawl(url: string, rules: RobotsRules, mode: RobotsMode): RobotsDecision {
  const path = new URL(url).pathname
  const disallowedMatch = rules.disallowed.find(d => path.startsWith(d))
  const allowMatch = rules.explicit_allows.find(a => path.startsWith(a))
  const blocked = !!disallowedMatch && !(allowMatch && allowMatch.length >= disallowedMatch.length)

  if (mode === 'ignore') return { allowed: true, warnings: [] }
  if (mode === 'strict') return { allowed: !blocked, warnings: blocked ? [`robots.txt disallows ${path}`] : [] }
  // warn_but_allow (default per D-07)
  return { allowed: true, warnings: blocked ? [`robots.txt disallows ${path} — proceeding under agent-of-record assumption`] : [] }
}
```

import 'server-only'
import { extractFromPastedText, type ExtractedFieldRecord } from './profileExtractClaude'
import { SOURCE_CONFIG } from './profileConfig'
import { estimateCost } from './profileCostEstimate'
import { refuseIfInternalIp, SSRFError } from './profileWebsiteSSRFGuard'
import { fetchRobots, isAllowedForCrawl, type RobotsMode } from './profileWebsiteRobots'

export type CrawlScope = 'A' | 'B' | 'C'

export type CrawlArgs = {
  url: string
  agencyId: string
  clientId: string
  scope?: CrawlScope             // default 'A'
  useJs?: boolean                // default true unless ENABLE_PLAYWRIGHT=false
  robotsMode?: RobotsMode        // default 'warn_but_allow'
  costCap?: number               // default SOURCE_CONFIG.website_scrape.default_cost_cap = 1.50
}

export type CrawlResult = {
  records: ExtractedFieldRecord[]
  pages_crawled: string[]
  pages_skipped: Array<{ url: string; reason: string }>
  warnings: string[]
  cost_spent_usd: number
  aborted: boolean
  abort_reason?: 'cost_cap' | null
}

const TARGETED_PATHS = [
  '/', '/about', '/about-us', '/services', '/what-we-do', '/contact',
  '/contact-us', '/locations', '/team', '/our-team',
]
const PRIORITIZED_KEYWORDS = /about|service|contact|location|team|who-we|company/i
const MODE_A_CAP = 10
const MODE_B_CAP = 100
const MODE_B_DEPTH = 2
const MODE_C_CAP = 15
const PER_PAGE_COST = 0.011   // matches estimateCost website formula
const JS_OVERHEAD = 0.004

export async function crawlWebsite(args: CrawlArgs): Promise<CrawlResult> {
  const scope = args.scope ?? 'A'
  const useJs = args.useJs ?? (process.env.ENABLE_PLAYWRIGHT !== 'false')
  const robotsMode = args.robotsMode ?? 'warn_but_allow'
  const costCap = args.costCap ?? SOURCE_CONFIG.website_scrape.default_cost_cap

  const origin = new URL(args.url).origin
  const ceiling = SOURCE_CONFIG.website_scrape.confidence_ceiling

  const robots = await fetchRobots(args.url)
  const pagesSkipped: CrawlResult['pages_skipped'] = []
  const warnings: string[] = []
  let costSpent = 0
  let aborted = false
  let abortReason: 'cost_cap' | null = null

  // Resolve page list per scope
  let pagesInScope: string[] = []
  if (scope === 'A') {
    pagesInScope = TARGETED_PATHS.map(p => `${origin}${p}`).slice(0, MODE_A_CAP)
  } else if (scope === 'C') {
    pagesInScope = await resolveSitemapPages(origin, MODE_C_CAP)
    if (!pagesInScope.length) pagesInScope = TARGETED_PATHS.map(p => `${origin}${p}`).slice(0, MODE_A_CAP)
  } else {
    // Mode B: seed with homepage + targeted paths, BFS
    pagesInScope = await bfsPages(origin, useJs, MODE_B_CAP, MODE_B_DEPTH)
  }

  const pagesCrawled: string[] = []
  const records: ExtractedFieldRecord[] = []

  // Shared browser for useJs=true (avoid cold-start per page)
  let browser: any = null
  if (useJs) {
    const { chromium: pw } = await import('playwright-core')
    const chromium = (await import('@sparticuz/chromium')).default
    browser = await pw.launch({ args: chromium.args, executablePath: await chromium.executablePath(), headless: chromium.headless })
  }

  try {
    for (const pageUrl of pagesInScope) {
      // SSRF guard + robots.txt + cost-cap check before every fetch
      try { await refuseIfInternalIp(pageUrl) }
      catch (err) {
        if (err instanceof SSRFError) {
          pagesSkipped.push({ url: pageUrl, reason: `ssrf:${err.code}` })
          continue
        }
        throw err
      }
      const decision = isAllowedForCrawl(pageUrl, robots, robotsMode)
      warnings.push(...decision.warnings)
      if (!decision.allowed) { pagesSkipped.push({ url: pageUrl, reason: 'robots' }); continue }

      const perPageCost = PER_PAGE_COST + (useJs ? JS_OVERHEAD : 0)
      if (costSpent + perPageCost > costCap) {
        aborted = true; abortReason = 'cost_cap'; break
      }

      let text: string | null = null
      try {
        if (useJs) {
          const page = await browser.newPage()
          await page.goto(pageUrl, { waitUntil: 'networkidle', timeout: 15_000 })
          text = (await page.evaluate(() => (document.body?.innerText ?? '').slice(0, 50_000))) as string
          await page.close()
        } else {
          const cheerio = await import('cheerio')
          const r = await fetch(pageUrl, { redirect: 'manual' })
          if (!r.ok) { pagesSkipped.push({ url: pageUrl, reason: `http_${r.status}` }); continue }
          const html = await r.text()
          const $ = cheerio.load(html)
          $('script, style, noscript, nav, footer').remove()
          text = $('body').text().slice(0, 50_000)
        }
      } catch (err: any) {
        pagesSkipped.push({ url: pageUrl, reason: `fetch_error:${err?.message ?? 'unknown'}` })
        continue
      }
      if (!text || text.length < 40) { pagesSkipped.push({ url: pageUrl, reason: 'empty_content' }); continue }

      const extracted = await extractFromPastedText({
        text, agencyId: args.agencyId, clientId: args.clientId,
        sourceLabel: 'website_scrape', sourceUrl: pageUrl,
      })
      // D-10 per-page provenance + D-09 confidence clamp
      for (const { field_name, record } of extracted) {
        records.push({
          field_name,
          record: {
            ...record,
            source_type: 'website_scrape' as const,
            source_url: pageUrl,
            confidence: Math.min(record.confidence, ceiling),
          },
        })
      }
      pagesCrawled.push(pageUrl)
      costSpent += perPageCost
    }
  } finally {
    if (browser) await browser.close().catch(() => {})
  }

  return { records, pages_crawled: pagesCrawled, pages_skipped: pagesSkipped, warnings, cost_spent_usd: Math.round(costSpent * 10000) / 10000, aborted, abort_reason: abortReason }
}

async function resolveSitemapPages(origin: string, cap: number): Promise<string[]> {
  try {
    await refuseIfInternalIp(`${origin}/sitemap.xml`)
    const r = await fetch(`${origin}/sitemap.xml`)
    if (!r.ok) return []
    const body = await r.text()
    const locs = [...body.matchAll(/<loc>([^<]+)<\/loc>/g)].map(m => m[1].trim())
    // Prioritize about/services/contact patterns first
    const prioritized = locs.filter(u => PRIORITIZED_KEYWORDS.test(u))
    const rest = locs.filter(u => !PRIORITIZED_KEYWORDS.test(u))
    return [...prioritized, ...rest].slice(0, cap)
  } catch { return [] }
}

async function bfsPages(origin: string, useJs: boolean, cap: number, depth: number): Promise<string[]> {
  // Minimal BFS — start at homepage + targeted paths, collect hrefs that share origin.
  // Full implementation is deferred to post-MVP; v1 returns targeted-paths + sitemap-top-N.
  const seeds = [origin + '/', ...TARGETED_PATHS.slice(1).map(p => origin + p)]
  const sitemap = await resolveSitemapPages(origin, cap)
  const uniq = [...new Set([...seeds, ...sitemap])]
  // Depth constraint respected by capping page count; true link-following is
  // a future enhancement — RESEARCH §4 notes mode B "3-5 min" budget per crawl.
  return uniq.slice(0, cap)
}
